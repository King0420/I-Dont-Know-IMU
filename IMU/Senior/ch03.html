function inorderTraversal( root ) {
   if (!root) return []
   let res = []
   let stack = []
   let cur = root
   while(cur || stack.length) {
       while(cur) {
           stack.push(cur)
           cur = cur.left
       }
       let node = stack.pop()
       res.push(node.val)
       cur = node.right
   }
   return res
}

function maxDepth( root ) {
  if (!root) return 0
  let res = 0, queue = [root]
  while(queue.length) {
    let len = queue.length
    while(len--) {
      let node = queue.shift()
      node.left && queue.push(node.left)
      node.right && queue.push(node.right)
    }
    res++
  }
  return res
}
